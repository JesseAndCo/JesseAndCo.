<script>
/* ---------------- DATA ---------------- */

let status = localStorage.getItem("status") || "No updates yet";
let cash = localStorage.getItem("cash") || "$not set";
let cashVisible = localStorage.getItem("cashVisible") === "true";

/* ---------------- LOAD ---------------- */

document.getElementById("status").innerText = status;
document.getElementById("cashText").innerText = cash;

if (cashVisible) {
  document.getElementById("cashBox").style.display = "block";
}

/* ---------------- ADMIN LOGIN ---------------- */

function login() {
  const code = document.getElementById("code").value;

  if (code === "082712") {
    document.getElementById("adminPanel").style.display = "block";
    alert("Admin unlocked");
  } else {
    alert("Wrong code");
  }
}

/* ---------------- STATUS ---------------- */

function updateStatus() {
  const newStatus = document.getElementById("newStatus").value;

  localStorage.setItem("status", newStatus);
  document.getElementById("status").innerText = newStatus;
}

/* ---------------- CASH APP ---------------- */

function setCash() {
  const val = document.getElementById("cashInput").value;

  localStorage.setItem("cash", val);
  document.getElementById("cashText").innerText = val;
}

function toggleCash() {
  cashVisible = !cashVisible;
  localStorage.setItem("cashVisible", cashVisible);

  document.getElementById("cashBox").style.display =
    cashVisible ? "block" : "none";
}

/* ---------------- EVENTS ---------------- */

let events = JSON.parse(localStorage.getItem("events") || "[]");

function renderEvents() {
  const list = document.getElementById("eventList");
  list.innerHTML = "";

  events.forEach(e => {
    const li = document.createElement("li");
    li.innerText = e;
    list.appendChild(li);
  });
}

function addEvent() {
  const text = document.getElementById("eventText").value;
  if (!text) return;

  events.push(text);
  localStorage.setItem("events", JSON.stringify(events));

  renderEvents();
}

renderEvents();
</script>
